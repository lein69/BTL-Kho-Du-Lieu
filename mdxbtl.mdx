       [Dim Product].[Ten San Pham].MEMBERS ON ROWS

       [Dim Date].[Nam].MEMBERS ON ROWS

//Doanh thu theo quý

       [Dim Date].[Quy].MEMBERS ON ROWS

       [Measures].[Doanh Thu] ON COLUMNS
